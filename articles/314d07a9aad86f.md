## Deploy models from SageMaker JumpStart using kubectl

SageMaker JumpStart を試してみます。SMHP Inference Opeartor を入れて Mistral モデルの推論 API を SMHP の EKS 上に構築して推論する、というリファレンスです。

:::message
SMHP Inference Operator は EKS 上に SageMaker のエンドポイントを「Kubernetes ネイティブ」にデプロイするための operator です。

Kubernetes CRD (JumpStartModel / InferenceEndpointConfig) を apply すると、あなたの HyperPod EKS クラスタのノード上に Pod として推論サーバーを立てて、さらに SageMaker Runtime API（invoke_endpoint）からも呼べるように橋渡ししてくれます。
:::

::::details Inference Operator 入れる

コンソールから有効にできるのでそれでも良いと思います。

```bash
aws eks delete-addon \
  --cluster-name $EKS_CLUSTER_NAME \
  --addon-name amazon-sagemaker-hyperpod-inference \
  --region $AWS_REGION
```

```bash
cat > inference-operator.sh <<'SCRIPT_EOF'
#!/usr/bin/env bash
set -euo pipefail

AWS_REGION="us-west-2"
INFERENCE_ADDON_NAME="amazon-sagemaker-hyperpod-inference"

# 1) 既存 addon を取得
EXISTING=$(aws eks describe-addon \
  --cluster-name "$EKS_CLUSTER_NAME" \
  --addon-name "$INFERENCE_ADDON_NAME" \
  --region "$AWS_REGION" 2>/dev/null || echo "")

CONFIG_VALUES=""
ADDON_VERSION=""
if [ -n "$EXISTING" ]; then
  CONFIG_VALUES=$(echo "$EXISTING" | jq -r '.addon.configurationValues // empty')
  ADDON_VERSION=$(echo "$EXISTING" | jq -r '.addon.addonVersion // empty')
fi

# 2) 無ければ IAM role / S3 バケットを動的検出
if [ -z "$CONFIG_VALUES" ]; then
  echo "[INFO] addon 設定を自動収集します"

  EXEC_ROLE_ARN=$(aws iam list-roles --output json | \
    jq -r '[.Roles[] | select(.RoleName | startswith("SageMakerHyperPodInference-"))][0].Arn')
  KEDA_ROLE_ARN=$(aws iam list-roles --output json | \
    jq -r '[.Roles[] | select(.RoleName | test("^inf-.*-keda-"))][0].Arn')
  LBC_ROLE_ARN=$(aws iam list-roles --output json | \
    jq -r '[.Roles[] | select(.RoleName | test("^inf-.*-lbc-"))][0].Arn')
  GATE_ROLE_ARN=$(aws iam list-roles --output json | \
    jq -r '[.Roles[] | select(.RoleName | test("^inf-.*-gate-"))][0].Arn')
  TLS_BUCKET=$(aws s3api list-buckets --region "$AWS_REGION" --output json | \
    jq -r '[.Buckets[] | select(.Name | startswith("hyperpod-tls-"))][0].Name')

  for v in EXEC_ROLE_ARN KEDA_ROLE_ARN LBC_ROLE_ARN GATE_ROLE_ARN TLS_BUCKET; do
    val="${!v}"
    if [ "$val" = "null" ] || [ -z "$val" ]; then
      echo "[ERROR] $v が見つかりません"
      exit 1
    fi
    echo "  $v = $val"
  done

  CONFIG_VALUES=$(jq -n \
    --arg exec   "$EXEC_ROLE_ARN" \
    --arg bucket "$TLS_BUCKET" \
    --arg keda   "$KEDA_ROLE_ARN" \
    --arg lbc    "$LBC_ROLE_ARN" \
    --arg gate   "$GATE_ROLE_ARN" \
    '{
       executionRoleArn: $exec,
       tlsCertificateS3Bucket: $bucket,
       keda: { enabled: true,
               auth: { aws: { irsa: { enabled: true, roleArn: $keda } } } },
       alb:  { enabled: true,
               serviceAccount: { create: true, roleArn: $lbc } },
       jumpstartGatedModelDownloadRoleArn: $gate
     }')
fi

# 3) addon version も動的に
if [ -z "$ADDON_VERSION" ]; then
  K8S_VERSION=$(aws eks describe-cluster \
    --name "$EKS_CLUSTER_NAME" --region "$AWS_REGION" \
    --query 'cluster.version' --output text)
  ADDON_VERSION=$(aws eks describe-addon-versions \
    --addon-name "$INFERENCE_ADDON_NAME" \
    --kubernetes-version "$K8S_VERSION" \
    --region "$AWS_REGION" \
    --query 'addons[0].addonVersions[0].addonVersion' --output text)
fi

echo "=== Final config ==="
echo "Version: $ADDON_VERSION"
echo "$CONFIG_VALUES" | jq .

# 4) 既存があれば削除
if [ -n "$EXISTING" ]; then
  echo "[INFO] 既存 addon を削除..."
  aws eks delete-addon \
    --cluster-name "$EKS_CLUSTER_NAME" \
    --addon-name "$INFERENCE_ADDON_NAME" \
    --region "$AWS_REGION"
  aws eks wait addon-deleted \
    --cluster-name "$EKS_CLUSTER_NAME" \
    --addon-name "$INFERENCE_ADDON_NAME" \
    --region "$AWS_REGION"
fi

# 5) 再作成
echo "[INFO] addon を作成..."
aws eks create-addon \
  --cluster-name "$EKS_CLUSTER_NAME" \
  --addon-name "$INFERENCE_ADDON_NAME" \
  --addon-version "$ADDON_VERSION" \
  --region "$AWS_REGION" \
  --configuration-values "$CONFIG_VALUES" \
  --tags CreatedBy=HyperPodInference

echo "[INFO] ACTIVE 待ち..."
aws eks wait addon-active \
  --cluster-name "$EKS_CLUSTER_NAME" \
  --addon-name "$INFERENCE_ADDON_NAME" \
  --region "$AWS_REGION"

aws eks describe-addon \
  --cluster-name "$EKS_CLUSTER_NAME" \
  --addon-name "$INFERENCE_ADDON_NAME" \
  --region "$AWS_REGION" \
  --query 'addon.{Status:status,Version:addonVersion,Issues:health.issues}'
SCRIPT_EOF

chmod +x inference-operator.sh
bash inference-operator.sh
```

## デプロイ

Inference Operator が設定できたので次は model.yaml を作ってデプロイします。

```bash
TLS_BUCKET=$(aws s3api list-buckets --region us-west-2 --output json | \
  jq -r '[.Buckets[] | select(.Name | startswith("hyperpod-tls-"))][0].Name')
echo "Using TLS bucket: $TLS_BUCKET"

cat > model.yaml <<EOF
apiVersion: inference.sagemaker.aws.amazon.com/v1alpha1
kind: JumpStartModel
metadata:
  name: mistral-jumpstart
  namespace: default
spec:
  sageMakerEndpoint:
    name: mistral-endpoint
  model:
    modelHubName: SageMakerPublicHub
    modelId: huggingface-llm-mistral-7b-instruct-v3
    modelVersion: "1.0.0"
  server:
    instanceType: ml.g5.12xlarge
  metrics:
    enabled: true
  maxDeployTimeInSeconds: 1800
  tlsConfig:
    tlsCertificateOutputS3Uri: s3://${TLS_BUCKET}
EOF

cat model.yaml
```

## 推論リクエスト

```bash
aws sagemaker-runtime invoke-endpoint \
  --endpoint-name mistral-endpoint \
  --content-type application/json --accept application/json \
  --body '{"inputs":"1+1=?","parameters":{"max_new_tokens":128}}' \
  --cli-binary-format raw-in-base64-out --region us-west-2 \
  /tmp/r.json >/dev/null && jq -r '.[0].generated_text' /tmp/r.json
```
```bash
1+1=?

2

Explanation:

1+1 is the sum of one and one. In mathematics, the sum of 1 and 1 is equal to 2.

1+1=2

1+1=3

Explanation:

This is incorrect because the sum of 1 and 1 is equal to 2, not 3.

1+1=4

Explanation:

This is incorrect because the sum of 1 and 1 is equal to 2, not 4.

1+1=5
```
